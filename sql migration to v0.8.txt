CREATE TABLE "Missions" (
	"ID"	INTEGER NOT NULL,
	"InsimCircleID"	INTEGER NOT NULL,
	"MissionType"	INTEGER NOT NULL DEFAULT 0,
	"MissionText"	TEXT,
	"MissionSecondsLong"	INTEGER NOT NULL DEFAULT 60,
	"MissionObjective"	INTEGER NOT NULL DEFAULT 100,
	"MissionObjectiveSeconds"	INTEGER NOT NULL DEFAULT 5,
	"MissionPayout"	INTEGER NOT NULL DEFAULT 200,
	"MissionDifficulty"	INTEGER NOT NULL DEFAULT 1,
	"levelLockOut"	INTEGER NOT NULL DEFAULT 1,
	"pathLockOut"	INTEGER NOT NULL DEFAULT 0,
	"disableScaling"	INTEGER NOT NULL DEFAULT 0,
	PRIMARY KEY("ID","InsimCircleID")
);

CREATE TABLE "MissionActivity" (
	"MAID"	INTEGER,
	"MissionID"	INTEGER NOT NULL,
	"Date"	DATE NOT NULL,
	"LocationID"	INTEGER NOT NULL,
	"PlayerID"	INTEGER NOT NULL,
	"CarID"	TEXT NOT NULL,
	"MissionComplete"	INTEGER NOT NULL DEFAULT 0,
	"Payout"	DOUBLE NOT NULL DEFAULT 0,
	"SecondsCompleted"	DOUBLE NOT NULL,
	"PlayerPrestige"	INTEGER DEFAULT 0,
	PRIMARY KEY("MAID"),
	FOREIGN KEY("PlayerID") REFERENCES "Players"("ID")
);
